@end
类的属性声明
@interface Person:NSObject
@property NSString *firstName;
@property MSString *lastName;
@end

@property NSNumber *yearOfBirth;

减号方法（普通方法又称对象方法）声明
@interface Person:NSObject
-(void)someMethod;
-(void)someMethodWithValue:(SomeType)value;
@end
加号方法（类方法，又称静态方法）声明
@interface Person:NSObject

+(id)string;
+(id)stringWithString:(NSString *)aString;
@end
